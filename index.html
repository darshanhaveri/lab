# Python script displaying facts about our universe
def get_space_facts():
    facts = [
        "A day on Venus is longer than a year on Venus.",
        "Jupiter has 95 officially recognized moons.",
        "The James Webb Space Telescope recently discovered six massive galaxies that challenge our understanding of the early universe."
    ]
    
    for index, fact in enumerate(facts):
        print(f"Fact {index + 1}: {fact}")

if __name__ == "__main__":
    get_space_facts()
